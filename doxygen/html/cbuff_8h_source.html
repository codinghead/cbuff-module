00110 /*******************************************************************************
00111 * New data type CBUFFOBJ
00112 * Description:
00113 *   Used as an object to hold the context of each defined buffer 
00114 * Structure elements:                                                    
00115 *   startOfBuffer               - pointer to buffer start address (user
00116 *                               assigned)         
00117 *   endOfBuffer                 - pointer to buffer end address + 1 (user
00118 *                               assigned)               
00119 *   inPointer                   - pointer to write-in point into buffer 
00120 *                               (user assigned)           
00121 *   outPointer                  - pointer to read-out point from buffer 
00122 *                               (user assigned)         
00123 *   bufferNumber                - number assigned to this buffer (module
00124 *                               assigned)
00125 *   localFlag                   - context for this buffer to note if buffer
00126 *                               is full or empty (module assigned)
00127 *   nextCircBufferObj           - Pointer to next buffer in the linked-list 
00128 *                               (module assigned)   
00129 *******************************************************************************/
00130 typedef struct CBUFFTYPE {
00131     CBUFF               * startOfBuffer;
00132     CBUFF               * endOfBuffer;
00133     CBUFF               * inPointer;
00134     CBUFF               * outPointer;
00135     CBUFFNUM              bufferNumber;
00136     unsigned char         localFlag;
00137     struct CBUFFTYPE    * nextCircBufferObj;
00138 } CBUFFOBJ;
