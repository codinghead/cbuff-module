227   /**  \addtogroup  CBUFFcreateDestroyFunctions  Create/Destroy  Functions
228   *  CBUFF  Module's  functions  used  to  create  and  destroy  buffer  objects
229   *  @{ 
230   *******************************************************************************/
231   /******************************************************************************/
232   /**  \cond
233   *  Function  -  cbuffCreate()
234   *  \endcond
235   *
236   *  \brief  Creates  a  new  buffer  object  and  adds  it  to  the  global  linked  list  of 
237   *  buffers.
238   *
239   *  New  buffer  object  will  be  added  to  the  global  linked  list  of  buffers  providing
240   *  that  there  is  space  for  another  buffer.  In  total  16  buffers  can  be  supported
241   *  simultaneously.
242   *
243   *  \sa  cbuffDestroy()
244   *
245   *  \param  buffer              -  a  buffer  defined  by  the  caller
246   *  \param  sizeOfBuffer        -  size  of  aformentioned  buffer  in  bytes
247   *  \param  newCircBufferObj    -  buffer  object  to  insert  into  linked  list  of  buffer
248   *                                                 objects
249   *
250   *  \retval      0  -  failed  to  create  the  buffer
251   *  \retval   &rsaquo =1  -  the  number  asigned  to  the  buffer  created
252   *
253   *  \note
254   *  -#  The  CBUFF  module  can  only  handle  up  to  a  maximum  of  16  buffers
255   *  -#  It  is  recommended  that  the  'sizeOfBuffer'  should  always  be  at  least  3  or
256   *     greater  to  be  useful.  This  will,  however,  not  be  checked  by  this  function.
257   *******************************************************************************/
258   CBUFFNUM    cbuffCreate(CBUFF         *  buffer,
259                           unsigned  int    sizeOfBuffer,
260                           CBUFFOBJ      *  newCircBufferObj)
261   {
262        CBUFFOBJ  *  localCircBufferObj;
263        CBUFFNUM  circBuffNum  =  0x0001;       /*  Assign  starting  buffer  number,  1,   */
264                                            /*  for  use  when  searching  for  the      */
265                                            /*  next  free  buffer  number             */
266                                            /*  0  reserved  for  alloc  failed!        */
267  
268                                            /*  Ensure  there  is  a  buffer  and  it     */
269                                            /*  has  size                            */
270        if  ((buffer  !=  (unsigned  char  *)  0)  &&  (sizeOfBuffer  !=  0))
271        {
272                                            /*  Ensure  we  got  somewhere  to  create   */
273                                            /*  our  buffer  object                   */
274            if  (newCircBufferObj  !=  (CBUFFOBJ  *)  0)
275            {
276                                            /*  Find  place  for  buffer  object  in     */
277                                            /*  the  linked  list  and  assign  number   */
278                                            /*  If  there  is  nothing  in  the  list,    */
279                                            /*  just  put  this  object  in  the  first   */
280                                            /*  position                            */
281                if  (activeCbuffObjs  ==  0  &&  startOfCbuffObjs  ==  (CBUFFOBJ  *)  0)
282                {
283                                            /*  Insert  buffer  object  at  start  of    */
284                                            /*  list                                */
285                    startOfCbuffObjs  =  newCircBufferObj;
286                                            /*  Setup  buffer  object                 */
287                    startOfCbuffObjs-&rsaquo startOfBuffer  =  buffer;
288                    startOfCbuffObjs-&rsaquo endOfBuffer    =  buffer  +  (sizeof(CBUFF)  *
289                                                  (sizeOfBuffer  -  1));
290                    startOfCbuffObjs-&rsaquo inPointer   =  buffer;
291                    startOfCbuffObjs-&rsaquo outPointer  =  buffer;
292                                            /*  Clear  all  flags                     */
293                    startOfCbuffObjs-&rsaquo localFlag  =  0x00;
294                                            /*  Set  buffer  empty  flag               */
295                    startOfCbuffObjs-&rsaquo localFlag  |=  CBUFF_EMPTY;
296                                            /*  Ensure  we  point  to  NULL             */
297                    startOfCbuffObjs-&rsaquo nextCircBufferObj  =  (CBUFFOBJ  *)  0;
298                                            /*  Assign  it  a  number                  */
299                    activeCbuffObjs  |=  circBuffNum;
300                    startOfCbuffObjs-&rsaquo bufferNumber  =  circBuffNum;
301                                            /*  Return  buffer  number                */
302                    return  startOfCbuffObjs-&rsaquo bufferNumber;
303                }
304                                            /*  Otherwise,  if  we  haven't  allocated  */
305                                            /*  all  the  cbuff  objects  we  can        */
306                                            /*  support,  insert  another             */
307                else  if  (activeCbuffObjs  !=  0xFFFF)
308                {
309                                            /*  Insert  this  object  at  the  start  of  */
310                                            /*  the  list                            */
311                    localCircBufferObj  =  startOfCbuffObjs;
312                    startOfCbuffObjs  =  newCircBufferObj;
313                                            /*  Setup  buffer  object                 */
314                    startOfCbuffObjs-&rsaquo startOfBuffer  =  buffer;
315                    startOfCbuffObjs-&rsaquo endOfBuffer    =  buffer  +  (sizeof(CBUFF)  *
316                                                      (sizeOfBuffer  -  1));
317                    startOfCbuffObjs-&rsaquo inPointer   =  buffer;
318                    startOfCbuffObjs-&rsaquo outPointer  =  buffer;
319                                            /*  Clear  all  flags                     */
320                    startOfCbuffObjs-&rsaquo localFlag  =  0x00;
321                                            /*  Set  buffer  empty  flag               */
322                    startOfCbuffObjs-&rsaquo localFlag  |=  CBUFF_EMPTY;
323                                            /*  Ensure  we  point  to  next  buffer      */
324                                            /*  object                              */
325                    startOfCbuffObjs-&rsaquo nextCircBufferObj  =  localCircBufferObj;
326                                            /*  Find  a  free  cbuff  number  for  this   */
327                                            /*  buffer                              */
328                                            /*  Cbuff  Number  '1'  is  assigned,  so    */
329                                            /*  start  looking  from  '2'  for  a  free   */
330                                            /*  number                              */
331                    circBuffNum &lsaquo &lsaquo =  1;
332                   
333                    do
334                    {
335                        if  (!(activeCbuffObjs  &  circBuffNum))
336                        {
337                                            /*  Assign  the  buffer  number            */
338                            activeCbuffObjs  |=  circBuffNum;
339                            startOfCbuffObjs-&rsaquo bufferNumber  =  circBuffNum;
340                            return  startOfCbuffObjs-&rsaquo bufferNumber;
341                        }
342                                            /*  That  wasn't  free;  try  next  bit      */
343                        circBuffNum &lsaquo &lsaquo =  1;
344                                            /*  Mask  just  in  case  unsigned  int  is   */
345                                            /*  bigger  than  16-bits                 */
346                        circBuffNum  &=  0xFFFF;
347                    }  while(circBuffNum  !=  0x0000);
348                }
349            }
350           }
351                                            /*  If  buffer  allocation  failed,        */
352                                            /*  return  0  to  callee                  */
353        return  0;
354   }

356   /******************************************************************************/
357   /**  \cond
358   *  Function  -  cbuffDestroy()
359   *  \endcond
360   *
361   *  \brief  Destroys  an  exisiting  buffer  object  and  removes  it  to  the  global  linked
362   *  list  of  buffers.
363   *
364   *  The  buffer  object  will  be  removed  from  the  global  linked  list  of  buffers,
365   *  providing  that  the  buffer  exists  and  that  the  buffer  has  been  closed.
366   *
367   *  \sa  cbuffCreate()
368   *
369   *  \param  bufferNumber        -  number  of  buffer  to  destroy  as  returned  by
370   *                              cbuffClose()
371   *
372   *  \retval    CBUFF_DESTROY_FAIL   -  failed  to  destroy  requested  object
373   *  \retval    CBUFF_DESTROY_OK     -  destroyed  requested  object  successfully
374   *
375   *  \note
376   *  -#  Caller  must  have  'created'  (cbuffCreate())  at  least  one  circular  buffer
377   *    object  before  calling  this  function
378   *  -#  Destroying  a  buffer  object  does  not  delete  the  buffer's  content,  nor  free
379   *    the  associated  CBUFFOBJ  memory  used
380   *  -#  An  open  buffer  cannot  be  destroyed.  Such  a  case  returns  a  FAIL
381   *******************************************************************************/
382   unsigned  char  cbuffDestroy(CBUFFNUM  bufferNumber)
383   {
384        CBUFFOBJ  *  localCircBufferObj;
385        CBUFFOBJ  *  previousCircBufferObj;
386                                            /*  Make  sure  there  is  something  to     */
387                                            /*  remove                              */
388        if  (startOfCbuffObjs  !=  (CBUFFOBJ  *)  0)
389        {
390                                            /*  Check  if  first  object  is  the  one    */
391                                            /*  we  are  looking  for                  */
392            if  (startOfCbuffObjs-&rsaquo bufferNumber  ==  bufferNumber)
393            {
394                                            /*  Copy  where  this  object  is  pointing  */
395                                            /*  to  into  startOfCbuffObjs            */
396                startOfCbuffObjs  =  startOfCbuffObjs-&rsaquo nextCircBufferObj;
397                                            /*  Clear  this  buffers  bit  in  the       */
398                                            /*  active  buffers  variable             */
399                activeCbuffObjs  &=  ~bufferNumber;
400                                            /*  Destroyed  the  desired  object        */
401                return  CBUFF_DESTROY_OK;
402            }
403                                            /*  Otherwise  loop  through  all  objects  */
404                                            /*  if  there  are  some                   */
405            else  if  (startOfCbuffObjs-&rsaquo nextCircBufferObj  !=  (CBUFFOBJ  *)  0)
406            {
407                                            /*  Keep  a  copy  of  previous  object      */
408                                            /*  pointer                             */
409                previousCircBufferObj  =  startOfCbuffObjs;
410                                            /*  Get  next  object  in  the  list         */
411                localCircBufferObj  =  startOfCbuffObjs-&rsaquo nextCircBufferObj;
412                                            /*  Loop  through  each  item  and  check    */
413                do
414                {
415                    if  (localCircBufferObj-&rsaquo bufferNumber  ==  bufferNumber)
416                    {
417                                            /*  Remove  this  buffer  from  list        */
418                        previousCircBufferObj-&rsaquo nextCircBufferObj  = 
419                                              localCircBufferObj-&rsaquo nextCircBufferObj;
420                                            /*  Clear  this  buffers  bit  in  the       */
421                                            /*  active  buffers  variable             */
422                        activeCbuffObjs  &=  ~bufferNumber;
423                                            /*  Destroyed  the  desired  object        */
424                        return  CBUFF_DESTROY_OK;
425                    }
426                    else
427                    {
428                                            /*  That  wasn't  it                      */
429                                            /*  Move  to  next  object  in  the  list     */
430                        previousCircBufferObj  =  localCircBufferObj;
431                        localCircBufferObj  =  localCircBufferObj-&rsaquo nextCircBufferObj;
432                    }
433                }  while  (localCircBufferObj  !=  (CBUFFOBJ  *)  0);
434            }
435           
436        }
437                                            /*  Couldn't  find  the  desired  object    */
438        return  CBUFF_DESTROY_FAIL;
439   }

444   /**  \addtogroup  CBUFFopenCLoseFunctions  Open/Close  Functions
445   *  CBUFF  Module's  functions  used  to  open  and  close  buffer  objects  that  have  been
446   *  created.
447   *  @{ 
448   *******************************************************************************/
449   /******************************************************************************/
450   /**  \cond
451   *  Function  -  cbuffOpen()
452   *  \endcond
453   *
454   *  \brief  Opens  a  circular  buffer  for  use  by  caller  and  initialises  an  HCBUFFOBJ
455   *  handle  to  it
456   *
457   *  The  buffer  requested  will  be  initialised  for  first  use.  The  buffer  requested
458   *  by  bufferNumber  must  exist,  otherwise  this  function  will  fail.
459   *
460   *  \sa  cbuffClose()
461   *
462   *  \param  bufferNumber        -  number  of  an  existing  buffer  to  use
463   *
464   *  \retval    NULL        -  if  buffer  couldn't  be  created
465   *  \retval    handle   -  if  buffer  was  created  properly
466   *
467   *  \note
468   *  -#  Caller  must  have  created  (cbuffCreate())  at  least  one  circular  buffer
469   *     object  before  calling  this  function
470   *******************************************************************************/
471   HCBUFF  cbuffOpen(CBUFFNUM  bufferNumber)
472   {
473           CBUFFOBJ  *  localCircBufferObj;
474  
475                                            /*  Check  that  a  CBUFFNUM  object     */
476                                            /*  exists  with  bufferNumber            */
477        localCircBufferObj  =  startOfCbuffObjs;
478  
479        do
480        {
481                if  (localCircBufferObj-&rsaquo bufferNumber  ==  bufferNumber)
482                {
483                                             /*  Check  buffer  is  not  already  open    */
